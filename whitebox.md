static int binary_search_start(int *a, int size, int target)
{
    int low = -1;
    int high = size;
    while (low + 1 < high) {
        int mid = low + (high - low) / 2;
        if (target > a[mid]) {
            low = mid;
        } else {
            high = mid;
        }
    }

    if (high == size || a[high] != target) {
        return -1;
    } else {
        return high;
    }
}
